üöÄ Method 1: Add Script to HTML Head (Easiest)

Step 1: Add this script tag to your website's HTML head section:
<script src="https://cdnjs.cloudflare.com/ajax/libs/gsap/3.12.2/gsap.min.js"></script> <script> // MyDayOne Scattered Images Hover Effect (function() { 'use strict'; console.log('üöÄ MyDayOne Hover Effect Loading...'); const CONFIG = { imageUrl: 'https://i.ibb.co/wvnTg2d/OIP.jpg', imageCount: 4, positions: [ { x: -200, y: -100 }, // Top left { x: 150, y: -80 }, // Top right { x: -180, y: 120 }, // Bottom left { x: 180, y: 100 } // Bottom right ] }; // CSS Styles const styles = ` .mydayone-hover-image { position: fixed; width: 150px; height: 200px; pointer-events: none; opacity: 0; z-index: 9999; border-radius: 12px; overflow: hidden; box-shadow: 0 20px 40px rgba(0, 0, 0, 0.3); background-size: cover; background-position: center center; background-repeat: no-repeat; background-image: url('${CONFIG.imageUrl}'); } .mydayone-hover-image-1 { transform: rotate(-8deg); width: 140px; height: 180px; } .mydayone-hover-image-2 { transform: rotate(5deg); width: 160px; height: 220px; } .mydayone-hover-image-3 { transform: rotate(-3deg); width: 130px; height: 170px; } .mydayone-hover-image-4 { transform: rotate(7deg); width: 155px; height: 210px; } `; function addStyles() { const styleSheet = document.createElement('style'); styleSheet.textContent = styles; document.head.appendChild(styleSheet); } function createHoverImages() { const images = []; for (let i = 0; i < CONFIG.imageCount; i++) { const img = document.createElement('div'); img.className = `mydayone-hover-image mydayone-hover-image-${i + 1}`; document.body.appendChild(img); images.push(img); } return images; } function findMyDayOneElements() { const selectors = [ 'img[alt*="My DayOne"]', 'img[alt*="MyDayOne"]', '[href="/"]', 'a[href="/"]' ]; const elements = []; selectors.forEach(selector => { try { document.querySelectorAll(selector).forEach(el => { if (!elements.includes(el)) elements.push(el); }); } catch (e) {} }); return elements; } function addHoverEffect(element, hoverImages) { element.addEventListener('mouseenter', function(e) { hoverImages.forEach((img, index) => { const pos = CONFIG.positions[index]; img.style.left = (e.clientX + pos.x) + 'px'; img.style.top = (e.clientY + pos.y) + 'px'; gsap.fromTo(img, { opacity: 0, scale: 0.5, rotation: (index - 2) * 20 }, { duration: 0.6 + (index * 0.1), opacity: 1, scale: 1, rotation: (index - 2) * 8, ease: "back.out(1.7)", delay: index * 0.1 }); }); }); element.addEventListener('mousemove', function(e) { hoverImages.forEach((img, index) => { const pos = CONFIG.positions[index]; gsap.to(img, { duration: 0.3, left: (e.clientX + pos.x) + 'px', top: (e.clientY + pos.y) + 'px', ease: "power2.out" }); }); }); element.addEventListener('mouseleave', function() { hoverImages.forEach((img, index) => { gsap.to(img, { duration: 0.4, opacity: 0, scale: 0.3, rotation: (index - 2) * 30, ease: "power2.in", delay: index * 0.05 }); }); }); } function init() { addStyles(); const hoverImages = createHoverImages(); const elements = findMyDayOneElements(); if (elements.length === 0) { setTimeout(init, 2000); return; } elements.forEach(element => addHoverEffect(element, hoverImages)); console.log(`‚úÖ Hover effect added to ${elements.length} elements!`); } if (document.readyState === 'loading') { document.addEventListener('DOMContentLoaded', init); } else { init(); } setTimeout(init, 1000); setTimeout(init, 3000); })(); </script>
